Verify Oracle 23c Free Edition:
  SELECT BANNER FROM V$VERSION WHERE BANNER LIKE 'Oracle%';

Test BASIC Compression:
  CREATE TABLE test AS SELECT * FROM all_objects;
  ALTER TABLE test MOVE COMPRESS BASIC;

Test OLTP Compression:
  ALTER TABLE test MOVE COMPRESS FOR OLTP;

Check Available Compression:
  SELECT DISTINCT compression FROM dba_tables
  WHERE owner = USER;

Install HCC Compression Advisor:
  @sql/install_full.sql

Run Analysis:
  EXEC PKG_COMPRESSION_ADVISOR.run_analysis(NULL, 2, 4);

View Recommendations:
  SELECT * FROM T_COMPRESSION_ANALYSIS
  WHERE advisable_compression != 'NONE'
  ORDER BY projected_savings_mb DESC;

Execute Compression:
  EXEC PKG_COMPRESSION_EXECUTOR.compress_table(
    'MYSCHEMA', 'MYTABLE', 'OLTP', TRUE, FALSE
  );
